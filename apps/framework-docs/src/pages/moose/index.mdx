#### Data Ingestion
```ts filename="Receive & Validate Events from Data Source" copy
// Define your data model
interface Vote {
  postId: Key<string>;
  userId: string;
  voteType: 'up' | 'down';
  timestamp: Date;
}

// Create pipeline with HTTP API + Stream (no table yet)
export const votes = new IngestPipeline<Vote>("votes", {
  ingest: true,    // ✅ HTTP API: POST /ingest/votes
  stream: true,    // ✅ Redpanda stream for real-time processing
  table: false     // ❌ Skip storage for raw votes
});
```

#### Real-time Stream Processing
```ts filename="Process Events in Real-Time" copy
// Define processed vote model
interface ProcessedVote extends Vote {
  score: number;
  hour: Date;
}

// Create destination stream + table for processed votes
export const processedVotes = new IngestPipeline<ProcessedVote>("processed_votes", {
  ingest: false,   // ❌ No direct ingestion
  stream: true,    // ✅ Stream for further processing
  table: true      // ✅ Store processed votes
});

// Transform raw votes into scored votes
votes.stream!.addTransform(processedVotes.stream!, (vote: Vote) => ({
  ...vote,
  score: vote.voteType === 'up' ? 1 : -1,
  hour: new Date(vote.timestamp.getFullYear(), vote.timestamp.getMonth(), 
                 vote.timestamp.getDate(), vote.timestamp.getHours())
}));
```

#### In-Database Aggregation
```ts filename="Create Real-Time Leaderboard" copy
// Define leaderboard entry model
interface LeaderboardEntry {
  postId: string;
  totalScore: number;
  totalVotes: number;
  hour: Date;
}

// Create materialized view for real-time aggregation
export const leaderboard = new MaterializedView<LeaderboardEntry>({
  selectStatement: sql`
    SELECT 
      postId, 
      sum(score) as totalScore,
      count() as totalVotes,
      toStartOfHour(timestamp) as hour
    FROM ${processedVotes.table!} 
    GROUP BY postId, hour
    ORDER BY totalScore DESC
  `,
  selectTables: [processedVotes.table!],
  tableName: "post_leaderboard_table",
  materializedViewName: "post_leaderboard_mv"
});
```

#### Analytics API
```ts filename="Query Aggregated Data" copy
// Define API parameters
interface LeaderboardParams {
  limit?: number;
  minScore?: number;
}

// Create consumption API
export const getLeaderboard = new ConsumptionApi<
  LeaderboardParams, 
  LeaderboardEntry[]
>("leaderboard", async (params, { client }) => {
  const limit = params.limit || 10;
  const minScore = params.minScore || 0;
  
  return await client.query(`
    SELECT postId, totalScore, totalVotes, hour
    FROM post_leaderboard_table 
    WHERE totalScore >= ${minScore}
    ORDER BY totalScore DESC 
    LIMIT ${limit}
  `);
});
```

```python filename="Receive & Validate Events from Data Source" copy
# Define your data model
class Vote(BaseModel):

    userId: str
    voteType: Literal["up", "down"]
    timestamp: datetime

votes = IngestPipeline[Vote]("votes", IngestPipelineConfig(
    ingest=True,    # ✅ HTTP API: POST /ingest/votes
    stream=True,    # ✅ Redpanda stream for real-time processing
    table=False     # ❌ Skip storage for raw votes
))
```

#### Real-time Stream Processing
```python filename="Process Events in Real-Time" copy
# Define processed vote model
class ProcessedVote(Vote):
    score: int
    hour: datetime

# Create destination stream + table for processed votes
processed_votes = IngestPipeline[ProcessedVote]("processed_votes", IngestPipelineConfig(
    ingest=False,   # ❌ No direct ingestion
    stream=True,    # ✅ Stream for further processing
    table=True      # ✅ Store processed votes
))

# Transform raw votes into scored votes
def transform_vote(vote: Vote) -> ProcessedVote:
    return ProcessedVote(
        **vote.model_dump(),
        score=1 if vote.voteType == "up" else -1,
        hour=vote.timestamp.replace(minute=0, second=0, microsecond=0)
    )

votes.get_stream().add_transform(processed_votes.get_stream(), transform_vote)
```

#### In-Database Aggregation
```python filename="Create Real-Time Leaderboard" copy
# Define leaderboard entry model
class LeaderboardEntry(BaseModel):
    postId: str
    totalScore: int
    totalVotes: int
    hour: datetime

# Create materialized view for real-time aggregation
leaderboard = MaterializedView[LeaderboardEntry](MaterializedViewOptions(
    select_statement="""
        SELECT 
          postId, 
          sum(score) as totalScore,
          count() as totalVotes,
          toStartOfHour(timestamp) as hour
        FROM processed_votes_table 
        GROUP BY postId, hour
        ORDER BY totalScore DESC
    """,
    select_tables=[processed_votes.get_table()],
    table_name="post_leaderboard_table",
    materialized_view_name="post_leaderboard_mv"
))
```

#### Analytics API
```python filename="Query Aggregated Data" copy
# Define API parameters
class LeaderboardParams(BaseModel):
    limit: int = Field(default=10)
    minScore: int = Field(default=0)

# Create consumption API
def get_leaderboard(params: LeaderboardParams, client: MooseClient):
    return client.query(f"""
        SELECT postId, totalScore, totalVotes, hour
        FROM post_leaderboard_table 
        WHERE totalScore >= {params.minScore}
        ORDER BY totalScore DESC 
        LIMIT {params.limit}
    """)

leaderboard_api = ConsumptionApi[LeaderboardParams, list[LeaderboardEntry]](
    "leaderboard", 
    get_leaderboard
)
```

```bash filename="Start your development environment" copy
moose dev

# ✅ HTTP APIs auto-generated from your models
# ✅ Hot-reload enabled for instant feedback
```
